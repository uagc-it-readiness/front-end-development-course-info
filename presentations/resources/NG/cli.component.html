Passing info into a component.

We use the [] on our attributes to tell Angular these are bindable properties.


import { Component } from '@angular/core';

@Component({
  selector: 'my-app',
  template: ``
})
export class AppComponent { 
    name = 'Angular'; 
    desc = "Lorem ipsum dolor sit amet.";
}

Accepting info into a component.


import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-my-new-comp',
  templateUrl: './my-new-comp.component.html',
  styleUrls: ['./my-new-comp.component.css']
})
export class MyNewCompComponent implements OnInit {
  @Input() title: string;
  @Input() body: string;
  constructor() { }

}
